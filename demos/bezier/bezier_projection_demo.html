// Create 3D curve
const curve3d = Snap_ia.bezier([
    {x: 50, y: 50, z: 0},
    {x: 150, y: 100, z: 100},
    {x: 250, y: 100, z: 50},
    {x: 350, y: 50, z: 0}
]);

// Project to 2D
const topView =
  curve3d.projectToPlane('xy');
const perspView =
  curve3d.projectPerspective({
    focalLength: 400,
    camera: {x: -200, y: 100, z: -100}
  });
